import React, { useState } from 'react';
import { Job } from '../types';
import { MapPin, DollarSign, Clock, Heart, Sparkles, X } from 'lucide-react';
import { analyzeJobMatch } from '../services/geminiService';

interface JobCardProps {
  job: Job;
}

export const JobCard: React.FC<JobCardProps> = ({ job }) => {
  const [isLiked, setIsLiked] = useState(false);
  const [showAiAnalysis, setShowAiAnalysis] = useState(false);
  const [aiContent, setAiContent] = useState<string>('');
  const [loadingAi, setLoadingAi] = useState(false);

  const handleAiAnalysis = async (e: React.MouseEvent) => {
    e.preventDefault();
    if (showAiAnalysis) {
      setShowAiAnalysis(false);
      return;
    }
    
    setLoadingAi(true);
    setShowAiAnalysis(true);
    const result = await analyzeJobMatch(job.title, job.company);
    setAiContent(result);
    setLoadingAi(false);
  };

  return (
    <div className="group bg-white rounded-2xl border border-gray-100 shadow-sm hover:shadow-xl hover:-translate-y-1 transition-all duration-300 relative overflow-hidden flex flex-col h-full">
      {/* Hot Badge */}
      {job.isHot && (
        <div className="absolute top-0 right-0 bg-red-500 text-white text-[10px] font-bold px-3 py-1 rounded-bl-xl z-10 shadow-sm">
          HOT
        </div>
      )}

      <div className="p-6 flex-grow">
        <div className="flex items-start justify-between mb-4">
          <div className="flex items-center space-x-3">
            <div className="w-12 h-12 rounded-xl bg-gray-50 border border-gray-200 p-2 flex items-center justify-center overflow-hidden">
               <img src={job.logo} alt={job.company} className="object-contain w-full h-full" />
            </div>
            <div>
              <h3 className="font-bold text-gray-900 group-hover:text-indigo-600 transition-colors line-clamp-1 text-lg">
                {job.title}
              </h3>
              <p className="text-sm text-gray-500 font-medium">{job.company}</p>
            </div>
          </div>
          <button 
            onClick={() => setIsLiked(!isLiked)}
            className={`p-2 rounded-full transition-colors ${isLiked ? 'bg-pink-50 text-pink-500' : 'text-gray-300 hover:bg-gray-50 hover:text-gray-500'}`}
          >
            <Heart size={20} fill={isLiked ? "currentColor" : "none"} />
          </button>
        </div>

        <div className="space-y-2 mb-4">
          <div className="flex items-center text-sm text-gray-600">
            <DollarSign size={16} className="mr-2 text-green-600" />
            <span className="font-semibold text-green-600">{job.salary}</span>
          </div>
          <div className="flex items-center text-sm text-gray-500">
            <MapPin size={16} className="mr-2" />
            {job.location}
          </div>
           <div className="flex items-center text-sm text-gray-400">
            <Clock size={16} className="mr-2" />
            {job.postedDate}
          </div>
        </div>

        <div className="flex flex-wrap gap-2 mb-4">
          {job.tags.slice(0, 3).map((tag, idx) => (
            <span key={idx} className="px-2.5 py-1 bg-indigo-50 text-indigo-700 text-xs font-medium rounded-md">
              {tag}
            </span>
          ))}
          {job.tags.length > 3 && (
            <span className="px-2.5 py-1 bg-gray-50 text-gray-500 text-xs font-medium rounded-md">
              +{job.tags.length - 3}
            </span>
          )}
        </div>
      </div>

      {/* AI Section */}
      <div className="px-6 pb-2">
         {showAiAnalysis && (
          <div className="mb-4 bg-gradient-to-br from-indigo-50 to-purple-50 rounded-xl p-3 border border-indigo-100 text-sm animate-in fade-in slide-in-from-bottom-2 relative">
             <button onClick={() => setShowAiAnalysis(false)} className="absolute top-2 right-2 text-indigo-300 hover:text-indigo-500">
                <X size={14} />
             </button>
             <div className="flex items-center space-x-2 text-indigo-700 font-bold mb-1">
                <Sparkles size={14} />
                <span>AI Insight</span>
             </div>
             {loadingAi ? (
                <div className="flex space-x-1 items-center h-6">
                   <div className="w-1.5 h-1.5 bg-indigo-400 rounded-full animate-bounce"></div>
                   <div className="w-1.5 h-1.5 bg-indigo-400 rounded-full animate-bounce delay-75"></div>
                   <div className="w-1.5 h-1.5 bg-indigo-400 rounded-full animate-bounce delay-150"></div>
                </div>
             ) : (
                <p className="text-gray-700 leading-relaxed text-xs">
                   {aiContent}
                </p>
             )}
          </div>
         )}
      </div>

      <div className="px-6 py-4 bg-gray-50 border-t border-gray-100 flex items-center justify-between">
         <button 
           onClick={handleAiAnalysis}
           className="text-xs font-semibold text-indigo-600 flex items-center hover:text-indigo-800 transition-colors"
         >
           <Sparkles size={14} className="mr-1" />
           {showAiAnalysis ? "Đóng AI" : "AI Phân tích"}
         </button>
         <button className="bg-indigo-600 hover:bg-indigo-700 text-white text-sm font-semibold px-4 py-2 rounded-lg transition-colors shadow-lg shadow-indigo-200">
           Ứng tuyển ngay
         </button>
      </div>
    </div>
  );
};

import React, { useState, useRef, useEffect } from 'react';

} from 'lucide-react';

interface UserDropdownProps {
  user: User;
  onLogout: () => void;
}

export const UserDropdown: React.FC<UserDropdownProps> = ({ user, onLogout }) => {
  const [isOpen, setIsOpen] = useState(false);
  const dropdownRef = useRef<HTMLDivElement>(null);

  // Close dropdown when clicking outside
  useEffect(() => {
    const handleClickOutside = (event: MouseEvent) => {
      if (dropdownRef.current && !dropdownRef.current.contains(event.target as Node)) {
        setIsOpen(false);
      }
    };
    document.addEventListener('mousedown', handleClickOutside);
    return () => document.removeEventListener('mousedown', handleClickOutside);
  }, []);

  return (
    <div className="relative" ref={dropdownRef}>
      <button 
        onClick={() => setIsOpen(!isOpen)}
        className="flex items-center space-x-3 bg-white/10 hover:bg-white/20 px-3 py-2 rounded-full transition-all border border-indigo-400/30 text-white"
